                    function reverse(message){
                        let reversedMessage = "";
                    
                        for (let index = message.length-1; index >=0; index--) {
                            
                            reversedMessage += message[index];
                            
                        }
                        return reversedMessage;
                    }
                    
                    function displayReversed(revMessage,message){
                        let results = document.getElementById("results");
                        results.classList.add("revMessage"); 
                        
                        let displayMessage = "";
                    
                        message = message.replace(/\s/g, '');
                        revMessage = revMessage.replace(/\s/g, '');
                    
                        if (revMessage.toLowerCase() == message.toLowerCase()) { 
                            displayMessage = "YEAH! " + message + " is a cool palindrome!";
                        } else {
                            displayMessage = message + " is not a cool palindrome.";
                        }
                        results.innerHTML = displayMessage;
                    }

The code is structured with several functions.

getValue()

Gets the value entered by the user.

reverse()

Considers the input value and reverses it using a decrementing for loop.

displayReversed()

Considers the result of the string reversal and compares the two strings, then decides if they are the same.
